import { AdminPageTop } from './top/admin-page-top.component';

  MdButtonModule,
  MdIconModule,
  MdInputModule,
  MdMenuModule,
  MdSidenavModule,
  MdToolbarModule
} from '@angular/material';
import { appNameReducer, navReducer } from '../common/reducer/nav-reducer';
import { StoreModule } from '@ngrx/store';

/**
 * Admin Module

 */

    AdminRoutingModule

  declarations: [
    AdminComponent,
    MenuComponent,
    MenuItemComponent,
    AdminPageTop
  ],
  providers: [
    AdminGuard
  ]
