import { Component, EventEmitter, Input, OnInit, Output } from '@angular/core';
import { AdminMenu } from '../../../common/model/admin-menu';
import { rotateAnimation, toggleAnimation } from '../../../common/animation/toggleAnimation';
import { Router } from '@angular/router';
/**
 * AdminMenu item component
 * Created by maxu0 on 2017/6/4.
 */
@Component({
  selector: 'admin-menu-item',
  templateUrl: './menu-item.component.html',
  styleUrls: ['./menu-item.component.scss'],
  animations: [
    toggleAnimation,
    rotateAnimation
  ]
})
export class MenuItemComponent implements OnInit {

  showMenu: boolean; // 

  @Input()
  menuItem: AdminMenu;

  constructor(private router: Router) {

  }

  ngOnInit(): void {
    this.showMenu = JSON.stringify(this.menuItem).includes(`"path":"${this.router.url}"`);
  }

  // hover
  @Output()
  itemHover = new EventEmitter<any>();

  // 
  toggleMenu(): void{
    this.showMenu = !this.showMenu;
  }

  // hover
  onHoverItem($event): void {
    this.itemHover.emit($event);
  }

}

<li>
    <a md-button *ngIf="menuItem.subMenu?.length; else linkItem" fxFlexFill (click)="toggleMenu()" (mouseenter)="onHoverItem($event)">

        <span>{{menuItem.name}}</span>
        <md-icon [@rotateAnimation]="showMenu ? 'down': 'right'" class="rotate-icon">chevron_right</md-icon>
    </a>
    <ng-template #linkItem>
        <a md-button [routerLink]="menuItem.path" fxFlexFill class="linkItem" (mouseenter)="onHoverItem($event)" routerLinkActive="active">
            <span>{{menuItem.name}}</span>
        </a>
    </ng-template>
    <ul *ngIf="menuItem.subMenu?.length && showMenu" class="aside-subMenu-ul" @toggleAnimation>
        <admin-menu-item
                *ngFor="let item of menuItem.subMenu"
                [menuItem]="item"
                (itemHover)="onHoverItem($event)">
        </admin-menu-item>
    </ul>
</li>
